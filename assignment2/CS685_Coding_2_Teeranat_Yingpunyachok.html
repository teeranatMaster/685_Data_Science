# TODO: Compute Pearson correlation matrix using numpy.corrcoef
fig, ax = plt.subplots(figsize=(10, 8))
correlation_matrix = np.corrcoef(data, rowvar=False)
# TODO: Plot heatmap using ax.imshow
cs_m = ax.imshow(correlation_matrix, cmap='coolwarm', vmin=-1, vmax=1)
# TODO: Add colorbar
fig.colorbar(cs_m)
# TODO: Set X and Y labels to the feature names
ax.set_xticks(np.arange(len(feature_names)))
ax.set_yticks(np.arange(len(feature_names)))
ax.set_xticklabels(feature_names,rotation=45,ha='right')
ax.set_yticklabels(feature_names)
for i in range(len(feature_names)):
    for j in range(len(feature_names)):
        label = np.round(correlation_matrix[i,j], decimals=2)
        ax.text(j, i, label, ha='center', va='center', color='black')

ax.set_title("Correlation Heatmap of Wine Features (pyplot only)")
fig.tight_layout()
plt.show()
